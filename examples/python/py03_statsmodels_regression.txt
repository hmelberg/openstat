# label: statsmodels — OLS regression
# Example: statsmodels — OLS regression with a formula
# Microdata fetches variables. statsmodels.formula.api runs OLS.

#micro
require no.ssb.fdb:54 as fd
create-dataset reg
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#python
import statsmodels.formula.api as smf

reg["income_thousand"] = reg["income"] / 1000
model = smf.ols("income_thousand ~ C(gender)", data=reg).fit()
model.summary()

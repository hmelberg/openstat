# Small dataset, describe() and groupby().
import pandas as pd

df = pd.DataFrame({
    "gender": ["Male", "Female", "Male", "Female", "Male", "Female"],
    "income": [620000, 540000, 710000, 820000, 480000, 610000],
    "age":    [34, 41, 29, 52, 38, 45],
})

df.describe()

df.groupby("gender")["income"].agg(["mean", "median", "count"])

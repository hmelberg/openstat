# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris

import ui

# The controls land in a strip at the top of the output area. The default
# in plain scripts is rerun="none": you change the values and press Run when you want.
n = ui.slider(1, 50, value=10, label="Number of rows", name="n")
species = ui.dropdown(["setosa", "versicolor", "virginica"], label="Species", name="species")

# rerun="all": when this changes, the WHOLE script reruns (debounced).
show_mean = ui.checkbox(label="Show mean", value=False, name="mean", rerun="all")

# sync_to: keeps the python variable `threshold` updated in the session
# without running anything — useful for ui.on handlers.
threshold = ui.slider(0, 10, value=5, step=0.5, label="Threshold", name="t", sync_to="threshold")

selection = iris[iris["species"] == species].head(n)
selection if not show_mean else selection.describe()

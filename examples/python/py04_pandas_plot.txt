# Example: pandas + matplotlib — histogram and boxplot
# `# load` fetches the data. matplotlib plots directly.
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/person_year_sample.csv as plot_data

import matplotlib.pyplot as plt

plot_data["income_thousand"] = plot_data["income"] / 1000

plot_data["income_thousand"].plot.hist(bins=30, edgecolor="white")
plt.xlabel("Income (thousand kr)")
plt.title("Distribution of wage income")
plt.show()

plot_data.boxplot(column="income_thousand", by="status", rot=45)
plt.suptitle("")
plt.title("Wage income by status")
plt.ylabel("Income (thousand kr)")
plt.show()

# Example: pandas + matplotlib — histogram and boxplot
# Microdata fetches variables. matplotlib plots directly.

#micro
require no.ssb.fdb:54 as fd
create-dataset plot_data
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#python
import matplotlib.pyplot as plt

plot_data["income_thousand"] = plot_data["income"] / 1000

plot_data["income_thousand"].plot.hist(bins=30, edgecolor="white")
plt.xlabel("Income (thousand kr)")
plt.title("Distribution of wage income")
plt.show()

plot_data.boxplot(column="income_thousand", by="gender")
plt.suptitle("")
plt.title("Wage income by gender")
plt.ylabel("Income (thousand kr)")
plt.show()

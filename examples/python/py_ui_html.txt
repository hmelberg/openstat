# `ui.html` — HTML from python

`ui.html.ELEMENT(...)` builds real DOM elements: positional arguments
are children (text or other elements), keywords are attributes
(`cls=`/`class_=`, `style=` as a string or dict, `data_x=` → `data-x`).
The last expression in a cell is mounted in the cell's slot — or use
`.show(target="an-id")` to place it in an html cell.

#%% python
import ui

card = ui.html.div(
    ui.html.h3("A card"),
    ui.html.p("Built with ", ui.html.b("ui.html"), "."),
    cls="card", style={"border": "1px solid #ccc", "padding": "12px", "border_radius": "8px"},
    data_source="ui-html-example",
)
card

`#tag.import = shoelace` in the preamble loaded Shoelace — so
`ui.sl.*` is now available as web components:

#%% python
import ui

counter = {"n": 0}
display = ui.html.span("0", id="counter-display")

def click(event):
    counter["n"] += 1
    display.clear()
    display.add(str(counter["n"]))

button = ui.sl.button("Click me", variant="primary")
button.on("click", click);
ui.html.div(button, ui.html.p("Click count: ", display))

#%% python
import ui

# Widget callback WITHOUT rerun: the handler gets the new value directly.
message = ui.html.div("Move the slider …")

def handle_change(v):
    message.clear()
    message.add("The value is now " + str(v) + " (read via ui.value: " + str(ui.value("s")) + ")")

ui.slider(0, 100, value=25, name="s", label="Try me", on_change=handle_change);
message

# Eksempel: dplyr — omkoding og nye variabler
# Viser case_when, ifelse og count. Oversettes til generate/replace/tabulate.

#micro
require no.ssb.fdb:54 as fd
create-dataset folk
import fd/BEFOLKNING_KJOENN as kjonn
import fd/INNTEKT_WLONN 2022-01-01 as inntekt
import fd/BEFOLKNING_FOEDSELS_AAR_MND as fodt

#r
library(dplyr)

folk <- folk |>
  mutate(
    alder      = 2022 - floor(fodt / 100),
    aldersgr   = case_when(alder < 30 ~ 1, alder < 60 ~ 2, TRUE ~ 3),
    hoyinntekt = ifelse(inntekt > 600000, 1, 0)
  )

folk |> count(aldersgr, hoyinntekt)

# Widgets in a MicroPython notebook

MicroPython runs on the main thread, so the `ui` module uses the same
pull model as python mode: each `ui.*` call reads the control's current
value, and changes rerun the cell (or a named cell via `rerun=`).

#%% micropython
import ui

n = ui.slider(1, 20, value=5, label="Number of rows")
species = ui.dropdown(sorted(set(iris["species"])), label="Species")

iris[iris["species"] == species].head(n)

#%% micropython id=plot
# This cell USES n/species from the cell above (shared session), but has
# no widgets of its own — the ui.run_button below runs it (target="plot").
import pandas_mpy as pd
import plotly_express_mpy as pe

sub = iris[iris["species"] == species].head(n)
pe.scatter(sub, x="sepal_length", y="petal_length",
           title="Iris — " + species + ", first " + str(n) + " rows")

#%% micropython
import ui

ui.run_button("Update", target="plot")

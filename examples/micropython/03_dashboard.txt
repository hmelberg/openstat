## Sales 2026

The same sales figures as the dash demo, now as `#%%` cells: the KPI
tiles sit in a single `cols=2` cell (`ui.kpi` + `show(...)` to mount
both cards in the same cell), and the two curve cards below each have
THEIR OWN region/cumulative control — `rerun="self"` (the default) means
each card reruns itself when its own control changes, just as
`d.add(curve, region=..., cumulative=...)` gave each dash card its own
widget strip.

#%% micropython
import plotly_express_mpy as pe

months = ["jan", "feb", "mar", "apr", "may", "jun"]
sales = {"North": [120, 135, 150, 160, 172, 190],
         "South": [200, 210, 190, 240, 260, 255]}

def curve(region, cumulative, title):
    values = sales[region]
    if cumulative:
        result = []
        total = 0
        for v in values:
            total += v
            result.append(total)
        values = result
    return pe.line(x=months, y=values, title=title)

#%% micropython id=kpi cols=2
import ui

show(ui.kpi(sum(sales["North"]) + sum(sales["South"]), unit="units", label="Total sales"),
     ui.kpi(max(sales["South"]), unit="units", label="Best month (South)"))

#%% micropython id=plot1
import ui

region = ui.dropdown(["North", "South"], label="Region", name="region1")
cumulative = ui.checkbox(label="Cumulative", value=False, name="cumulative1")
curve(region, cumulative, "Trend")

#%% micropython id=plot2
import ui

region = ui.dropdown(["North", "South"], label="Region", name="region2")
cumulative = ui.checkbox(label="Cumulative", value=False, name="cumulative2")
curve(region, cumulative, "Trend (copy with its own controls)")

# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris
import pandas_mpy as pd
import plotly_express_mpy as pe

# The frame is ready under the name you chose:
show(iris.head())

# A summary table — pick columns, then group:
show(iris[["species", "sepal_length", "petal_length"]].groupby("species").mean())

# An interactive scatter plot:
pe.scatter(iris, x="sepal_length", y="petal_length", color="species",
           title="Iris — sepal vs petal")

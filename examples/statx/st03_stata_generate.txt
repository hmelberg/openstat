# label: Stata — generate/egen
* Example: Stata – generate and egen
* The first block imports variables via microdata.
* The second block creates new variables: log income and mean income.

#micro
require no.ssb.fdb:54 as fd
create-dataset people
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#stata
generate log_income = log(income)
egen mean_income = mean(income)
summarize log_income mean_income

# Read from a LIVE public API — World Bank life expectancy (JSON).
from pyodide.http import open_url
import json, pandas as pd, matplotlib.pyplot as plt

url = ("https://api.worldbank.org/v2/country/NOR;SWE;DNK;FIN/"
       "indicator/SP.DYN.LE00.IN?format=json&per_page=400")
records = json.loads(open_url(url).read())[1]     # element 0 is pagination meta
life = pd.json_normalize(records)[["country.value", "date", "value"]].dropna()
life.columns = ["country", "year", "life_expectancy"]

latest = life.sort_values("year").groupby("country").last()
latest["life_expectancy"].round(1)                 # a table
latest["life_expectancy"].plot.bar(title="Life expectancy — latest year")
plt.ylabel("years"); plt.show()

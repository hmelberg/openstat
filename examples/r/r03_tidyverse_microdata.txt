# label: tidyverse + #micro
# Example: tidyverse on a microdata dataset
# The first block imports variables via microdata.
# The second block uses dplyr on the 'folk' dataset.

#micro
require no.ssb.fdb:54 as fd
create-dataset folk
import fd/BEFOLKNING_KJOENN as kjonn
import fd/INNTEKT_WLONN 2022-01-01 as inntekt

#r
# use folk
library(dplyr)

folk %>%
  group_by(kjonn) %>%
  summarise(
    snitt  = mean(inntekt, na.rm = TRUE),
    median = median(inntekt, na.rm = TRUE),
    antall = n()
  )

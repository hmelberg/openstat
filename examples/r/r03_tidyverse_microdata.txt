# label: tidyverse + #micro
# Example: tidyverse on a microdata dataset
# The first block imports variables via microdata.
# The second block uses dplyr on the 'folk' dataset.

#micro
require no.ssb.fdb:54 as fd
create-dataset folk
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#r
# use folk
library(dplyr)

folk %>%
  group_by(gender) %>%
  summarise(
    mean_income   = mean(income, na.rm = TRUE),
    median_income = median(income, na.rm = TRUE),
    n             = n()
  )

# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/person_year_sample.csv as folk

# Keep only people with positive wage income
adults <- subset(folk, !is.na(income) & income > 0)

# Create a new variable (income in thousands of kroner)
adults <- transform(adults, income_thousands = income / 1000)

# Mean income per status
aggregate(income ~ status, data = adults, FUN = mean)

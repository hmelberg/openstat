# label: dplyr — across + slice_sample
# Example: across() and random sampling
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/person_year_sample.csv as folk

library(dplyr)

# Scale several amount columns at once to thousands of kroner
folk <- folk |> mutate(across(c(income, total_income), ~ .x / 1000))

# Random sample — the seed gives a reproducible result
set.seed(2022)
sampled <- folk |> slice_sample(prop = 0.3)

sampled |>
  group_by(status) |>
  summarise(mean_income = mean(income, na.rm = TRUE))

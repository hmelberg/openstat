# label: tidyverse — group statistics
# Example: tidyverse — group statistics with dplyr
library(dplyr)

df <- tibble(
  gender = c("Male", "Female", "Male", "Female", "Male", "Female"),
  income = c(620000, 540000, 710000, 820000, 480000, 610000),
  age    = c(34, 41, 29, 52, 38, 45)
)

df %>%
  group_by(gender) %>%
  summarise(
    mean_income   = mean(income),
    median_income = median(income),
    n             = n()
  )

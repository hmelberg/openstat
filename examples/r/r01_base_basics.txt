# label: base R — basics
# Example: base R — frequency and mean
# table() for frequency, tapply() for group statistics.
df <- data.frame(
  gender = c("Male", "Female", "Male", "Female", "Male", "Female"),
  income = c(620000, 540000, 710000, 820000, 480000, 610000),
  age    = c(34, 41, 29, 52, 38, 45)
)

table(df$gender)
tapply(df$income, df$gender, mean)
summary(df$income)

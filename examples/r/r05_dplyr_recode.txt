# label: dplyr — recoding (case_when/ifelse)
# Example: dplyr — recoding and new variables
# Shows case_when, ifelse and count.
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/person_year_sample.csv as folk

library(dplyr)

folk <- folk |>
  mutate(
    age_group   = case_when(age < 30 ~ 1, age < 60 ~ 2, TRUE ~ 3),
    high_income = ifelse(!is.na(income) & income > 600000, 1, 0)
  )

folk |> count(age_group, high_income)

# label: dplyr — recoding (case_when/ifelse)
# Example: dplyr — recoding and new variables
# Shows case_when, ifelse and count. Translates to generate/replace/tabulate.

#micro
require no.ssb.fdb:54 as fd
create-dataset folk
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income
import fd/BEFOLKNING_FOEDSELS_AAR_MND as birth_ym

#r
# use folk
library(dplyr)

folk <- folk |>
  mutate(
    age         = 2022 - floor(birth_ym / 100),
    age_group   = case_when(age < 30 ~ 1, age < 60 ~ 2, TRUE ~ 3),
    high_income = ifelse(income > 600000, 1, 0)
  )

folk |> count(age_group, high_income)

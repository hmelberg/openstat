# label: regression — lm + glm
# Example: regression — linear and logistic
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/person_year_sample.csv as folk

library(dplyr)

# Linear regression of income on age and status
lm(income ~ age + factor(status), data = folk)

# Logistic regression: probability of high income
folk <- folk |> mutate(high_income = ifelse(!is.na(income) & income > 600000, 1, 0))
glm(high_income ~ age, family = binomial(), data = folk)

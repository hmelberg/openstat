# label: ggplot2 + #micro
# Example: ggplot2 — histogram and boxplot on a microdata dataset
# Microdata fetches variables. ggplot2 plots.

#micro
require no.ssb.fdb:54 as fd
create-dataset plottdata
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#r
# use plottdata
library(ggplot2)

plottdata$income_thousands <- plottdata$income / 1000

ggplot(plottdata, aes(x = income_thousands)) +
  geom_histogram(bins = 30, fill = "steelblue", colour = "white") +
  labs(x = "Income (thousand kr)", title = "Distribution of wage income")

ggplot(plottdata, aes(x = gender, y = income_thousands, fill = gender)) +
  geom_boxplot() +
  labs(y = "Income (thousand kr)", title = "Wage income by gender") +
  theme(legend.position = "none")

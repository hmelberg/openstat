const threshold = 5.0;
const filtered = iris.filter(aq.escape(d => d.sepal_length > threshold));
console.log(filtered.numRows() + " of " + iris.numRows() + " rows above the threshold");
#%%
// Builds on `filtered` from the previous cell:
filtered
  .groupby("species")
  .rollup({ n: op.count(), mean: op.mean("petal_length") })
#%%
// Last cell: Plotly figure of the filtered subset
({ data: [{
    x: Array.from(filtered.array("sepal_length")),
    y: Array.from(filtered.array("petal_length")),
    mode: "markers", type: "scatter",
    text: Array.from(filtered.array("species")),
  }],
  layout: { title: "Iris (sepal_length > " + threshold + ")",
            xaxis: { title: "sepal_length" }, yaxis: { title: "petal_length" } } })

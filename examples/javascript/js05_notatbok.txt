const terskel = 5.0;
const store = iris.filter(aq.escape(d => d.sepal_length > terskel));
console.log(store.numRows() + " av " + iris.numRows() + " rader over terskelen");
#%%
// Bygger videre på `store` fra forrige celle:
store
  .groupby("species")
  .rollup({ n: op.count(), snitt: op.mean("petal_length") })
#%%
// Siste celle: Plotly-figur av det filtrerte utvalget
({ data: [{
    x: Array.from(store.array("sepal_length")),
    y: Array.from(store.array("petal_length")),
    mode: "markers", type: "scatter",
    text: Array.from(store.array("species")),
  }],
  layout: { title: "Iris (sepal_length > " + terskel + ")",
            xaxis: { title: "sepal_length" }, yaxis: { title: "petal_length" } } })

// label: Regression and machine learning (ml.js)
// Example: linear regression (ss) and k-means clustering (ML)
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris

// Simple linear regression: petal_length ~ sepal_length
const pairs = iris.objects().map(d => [d.sepal_length, d.petal_length]);
const model = ss.linearRegression(pairs);
const rl = ss.linearRegressionLine(model);
console.log("petal_length =", model.b.toFixed(2), "+", model.m.toFixed(2), "* sepal_length");
console.log("r² =", ss.rSquared(pairs, rl).toFixed(3));
console.log("prediction at sepal_length=6:", rl(6).toFixed(2));

// k-means clustering with ml.js (ML also has PCA, random forest, KNN etc.):
const X = iris.objects().map(d => [d.sepal_length, d.sepal_width, d.petal_length, d.petal_width]);
const km = ML.KMeans(X, 3);

// How well do the clusters match the species? Cross-tabulation:
const withCluster = iris.derive({ row: aq.op.row_number() })
  .derive({ cluster: aq.escape(d => km.clusters[d.row - 1]) });
withCluster
  .groupby("species", "cluster")
  .count()
  .orderby("species", "cluster")

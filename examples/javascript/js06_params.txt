#%% md
# Parametre i JavaScript-celler

En tilordning med en `//@param`-kommentar får et skjema i cellen.
Cellen kjøres automatisk på nytt når verdien endres (skru av med `run:"manual"`).
NB: ingen semikolon på param-linjer — verdien byttes ut byte-nøyaktig.
#%%
terskel = 5  //@param {type:"slider", min:4, max:8, step:0.5}
art = 'setosa'  //@param {type:"string"}
inkluder_alle = false  //@param {type:"boolean", run:"manual"}  // krever klikk på Kjør-chippen

utvalg = inkluder_alle
  ? iris.filter(aq.escape(d => d.sepal_length > terskel))
  : iris.filter(aq.escape(d => d.sepal_length > terskel && d.species === art));
console.log(utvalg.numRows() + " rader (terskel=" + terskel +
            (inkluder_alle ? ", alle arter)" : ", art=" + art + ")"));
utvalg.groupby("species").rollup({ n: op.count(), snitt: op.mean("sepal_length") })

// label: Arquero — dataframes (filter, groupby, rollup)
// Example: Arquero tables — load CSV, filter, group and aggregate
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris

// The dataset is ready as an Arquero table under the name you chose:
console.log("rows:", iris.numRows(), "columns:", iris.numCols());

// Filter + new column (derive) + grouping + aggregation (rollup):
const summary = iris
  .filter(d => d.sepal_length > 4.5)
  .derive({ ratio: d => d.petal_length / d.sepal_length })
  .groupby("species")
  .rollup({
    n: op.count(),
    mean_sepal: op.mean("sepal_length"),
    mean_ratio: op.mean("ratio"),
  })
  .orderby(aq.desc("mean_sepal"));

// The last expression in the script is displayed automatically — tables as HTML:
summary

# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/penguins.csv as penguins

// Plotly: a {data, layout} object as the last expression becomes an interactive figure.
const speciesList = ["Adelie", "Chinstrap", "Gentoo"];
const meanWeights = speciesList.map(a =>
  ss.mean(Array.from(penguins.filter(aq.escape(d => d.species === a && d.body_mass_g != null))
                             .array("body_mass_g"))));
console.log("mean weight (g):", Object.fromEntries(speciesList.map((a, i) => [a, Math.round(meanWeights[i])])));

// Observable Plot: returns an SVG node — shown directly in the output.
// (Swap in the Plotly object below for an interactive version.)
Plot.dot(penguins.objects(), {
  x: "flipper_length_mm",
  y: "body_mass_g",
  stroke: "species",
}).plot({ grid: true })

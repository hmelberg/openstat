#micro
// Hybrid: build a dataset with a microdata import, then use DuckDB to make a smaller one.
require no.ssb.fdb:54 as fd
create-dataset folk
import fd/BEFOLKNING_KJOENN as gender
import fd/INNTEKT_WLONN 2022-01-01 as income

#duckdb
-- DuckDB/SQL uses -- for comments, not #.
CREATE OR REPLACE TABLE high_earners AS
SELECT * FROM folk WHERE income > 600000;

SELECT count(*) AS n, round(avg(income)) AS mean_income
FROM high_earners;

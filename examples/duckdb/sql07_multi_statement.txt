# label: SQL — multiple statements
-- Several statements in one script: create raw data, create a summary dataset,
-- and finally show a table. The statements run in order.

CREATE OR REPLACE TABLE people AS
SELECT * FROM (VALUES
  (1, 'Male',   25, 450000, 'Oslo'),
  (2, 'Female', 34, 620000, 'Bergen'),
  (3, 'Male',   52, 810000, 'Oslo'),
  (4, 'Female', 41, 530000, 'Trondheim'),
  (5, 'Male',   19, 210000, 'Bergen'),
  (6, 'Female', 67, 390000, 'Oslo')
) AS t(id, gender, age, income, municipality);

-- Summary dataset per municipality (appears in the data overview).
CREATE OR REPLACE TABLE by_municipality AS
SELECT municipality,
       count(*)           AS n,
       round(avg(income)) AS mean_income
FROM people
GROUP BY municipality;

-- The final SELECT is shown as a table in the output area.
SELECT * FROM by_municipality ORDER BY mean_income DESC;

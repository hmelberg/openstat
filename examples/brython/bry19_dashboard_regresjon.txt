#%% md
# What explains sepal length?

Choose a predictor. This cell uses the **push model** instead of cell
rerun: `on_change=` on the dropdown is a Python FUNCTION, not a list of
cell ids — the control itself then never reruns (`rerun=` has no effect),
the handler instead updates the plot, R², and model summary directly in
three pre-built `ui.html` slots (`.clear()` + `.add(...)`), just like the
button example in `py_ui_html.txt`.

#%% brython id=regresjon
import ui
import pandas_brython as pd
import plotly_express_brython as pe
import statsmodels.formula.api as smf

plot_slot = ui.html.div()
kpi_slot = ui.html.div()
table_slot = ui.html.div()

def _model(predictor):
    return smf.ols("sepal_length ~ " + predictor, iris).fit()

def _update(predictor):
    fig = pe.scatter(iris, x=predictor, y="sepal_length", color="species",
                      title="sepal_length vs " + predictor)
    m = _model(predictor)
    plot_slot.clear()
    plot_slot.add(ui.image(fig))
    kpi_slot.clear()
    kpi_slot.add(ui.kpi(round(m.rsquared, 3), label="R²"))
    table_slot.clear()
    table_slot.add(ui.html.pre(str(m.summary())))

predictor = ui.dropdown(["petal_length", "petal_width", "sepal_width"],
                         label="Predictor", name="predictor", on_change=_update)
_update(predictor)

ui.html.div(
    plot_slot,
    ui.html.div(kpi_slot, table_slot,
                style={"display": "flex", "gap": "16px", "flex_wrap": "wrap"}),
)

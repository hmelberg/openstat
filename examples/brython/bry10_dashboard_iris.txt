#%% md
# Iris dashboard

Choose a species and the minimum sepal length: the control cell sets
`on_change=["kpi", "plott", "tabell"]` (one list, THREE target cells) AND
`sync_to=` on each control — `sync_to` keeps `species`/`min_length` updated
in the session immediately, without the control cell itself rerunning.
`selection(...)` is defined in the first (control-free) cell below and
shared by all three target cells — the same trick as `curve(...)` in the
micropython variant (`03_dashboard.txt`).

#%% brython
import pandas_brython as pd

def selection(species, min_length):
    sub = iris[iris["species"] == species]
    return sub[sub["sepal_length"] >= min_length]

#%% brython id=kontroller
import ui

species = ui.dropdown(["setosa", "versicolor", "virginica"], label="Species", name="species",
                       on_change=["kpi", "plott", "tabell"], sync_to="species")
min_length = ui.slider(4, 8, value=4, step=1, label="Min. sepal length", name="min_length",
                        on_change=["kpi", "plott", "tabell"], sync_to="min_length")

#%% brython id=kpi cols=2
import ui

u = selection(species, min_length)
if len(u) == 0:
    show(ui.kpi("–", unit="pcs", label="Number of flowers"),
         ui.kpi("–", unit="cm", label="Mean sepal length"))
else:
    show(ui.kpi(len(u), unit="pcs", label="Number of flowers"),
         ui.kpi(round(u["sepal_length"].mean(), 2),
                unit="cm", label="Mean sepal length"))

#%% brython id=plott
import plotly_express_brython as pe

# Empty-selection guard: dragging min. sepal length to 8 (max) together with
# e.g. "setosa" (which never reaches 8) leaves the selection empty — mean()
# on an empty brython-pandas frame raises IndexError (a brython pandas quirk,
# not a real pandas bug). Same pattern as the kpi cell above: check len()
# BEFORE touching the frame contents, and show a friendly text message
# instead of a crashed cell — a small teaching moment about edge cases.
# NOTE: a plain string here (NOT ui.markdown(...)) is deliberate — both
# branches must go through the SAME display path (text/figure embedding via
# show(), never the ui.* element's own .show()/data-ui-shown mounting) so a
# rerun ALWAYS clears the previous run's figure/text before showing the new
# one (mixing display paths between two reruns of the same cell otherwise
# left a stale figure behind the new text — found during the 5b-4
# regression).
u = selection(species, min_length)
if len(u) == 0:
    show("No flowers match the selection — try a lower minimum.")
else:
    show(pe.histogram(u, x="sepal_length", title=species))

#%% brython id=tabell
selection(species, min_length).head(8)

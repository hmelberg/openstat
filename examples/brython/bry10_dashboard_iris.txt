#%% md
# Iris-dashboard

Velg art og minste sepal-lengde: kontrollcellen setter `on_change=["kpi",
"plott", "tabell"]` (én liste, TRE mål-celler) OG `sync_to=` på hver
kontroll — `sync_to` holder `art`/`min_lengde` oppdatert i sesjonen med det
samme, uten at kontrollcellen selv kjøres på nytt. `utvalg(...)` er
definert i den første (kontroll-løse) cellen under og deles av alle tre
mål-cellene — samme knep som `kurve(...)` i micropython-varianten
(`03_dashboard.txt`).

#%% brython
import pandas_brython as pd

def utvalg(art, min_lengde):
    sub = iris[iris["species"] == art]
    return sub[sub["sepal_length"] >= min_lengde]

#%% brython id=kontroller
import ui

art = ui.dropdown(["setosa", "versicolor", "virginica"], label="Art", name="art",
                   on_change=["kpi", "plott", "tabell"], sync_to="art")
min_lengde = ui.slider(4, 8, value=4, step=1, label="Min. sepal-lengde", name="min_lengde",
                        on_change=["kpi", "plott", "tabell"], sync_to="min_lengde")

#%% brython id=kpi cols=2
import ui

u = utvalg(art, min_lengde)
if len(u) == 0:
    show(ui.kpi("–", unit="stk", label="Antall blomster"),
         ui.kpi("–", unit="cm", label="Snitt sepal-lengde"))
else:
    show(ui.kpi(len(u), unit="stk", label="Antall blomster"),
         ui.kpi(round(u["sepal_length"].mean(), 2),
                unit="cm", label="Snitt sepal-lengde"))

#%% brython id=plott
import plotly_express_brython as pe

# Tom-utvalg-vaktpost: drar min. sepal-lengde til 8 (maks) sammen med f.eks.
# "setosa" (som aldri når 8) og utvalget blir tomt — mean() på en tom
# brython-pandas-frame kaster IndexError (brython sin pandas-kvirk, ikke en
# ekte pandas-feil). Samme mønster som kpi-cellen over: sjekk len() FØR vi
# rører frame-innholdet, og gi en vennlig tekst-melding i stedet for en
# krasjet celle — et lite undervisningsmoment om kantsaker. MERK: en ren
# streng her (IKKE ui.markdown(...)) med vilje — begge grener skal gå
# gjennom SAMME visningsvei (tekst/figur-innebygging via show(), aldri
# ui.*-elementets .show()/data-ui-shown-montering) slik at et rerun ALLTID
# tømmer forrige kjørings figur/tekst før den nye vises (blandede
# visningsveier mellom to reruns av samme celle etterlot ellers en
# gjenværende, utdatert figur bak den nye teksten — funnet under 5b-4-
# regresjonen).
u = utvalg(art, min_lengde)
if len(u) == 0:
    show("Ingen blomster matcher utvalget — prøv en lavere minstegrense.")
else:
    show(pe.histogram(u, x="sepal_length", title=art))

#%% brython id=tabell
utvalg(art, min_lengde).head(8)

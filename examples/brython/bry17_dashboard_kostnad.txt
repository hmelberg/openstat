# Budget impact — new drug

Three sliders drive three target cells at once: `on_change=["kpi", "kurve",
"tabell"]` (one list, THREE targets) + `sync_to=` on each of them. The
"Advanced" checkbox instead uses `ui.widget("uptake_pct")` — the HANDLE to
an already-declared control — and calls `.hide()`/`.show()` on it. That's a
push callback (no rerun): it only changes the visibility of the uptake
slider; the value itself lives on unchanged (`sync_to`) whether it's shown
or not.

#%% brython
YEARS = [1, 2, 3, 4, 5]

def cohort(price, patients, uptake_pct):
    """Simple uptake curve: the share is reached gradually over three years."""
    rows = []
    for t in YEARS:
        n = round(patients * min(1.0, (uptake_pct / 100) * t / 3))
        rows.append({"year": t, "patients": n, "mill_nok": round(n * price / 1e6, 2)})
    return rows

def cumulative(rows):
    """Deliberately local function scope (avoids leaking the loop variable
    into the session globals — see total_cost() below for the WHY details:
    exit-gate finding 2026-07-18, browser-verified)."""
    cum, s = [], 0.0
    for row in rows:
        s += row["mill_nok"]
        cum.append(round(s, 2))
    return cum

def total_cost(rows):
    """Sum of mill_nok over a list of rows. A REAL FUNCTION on purpose — NOT
    a bare generator/comprehension expression directly at cell top level.
    Exit-gate finding (2026-07-18): a bare generator expression at cell top
    level (e.g. `sum(row["mill_nok"] for row in rows)` right in a cell) can
    leak the loop variable `row` into _shared_vars via a Brython 3.12 dict
    trap (the same trap that brython_runner.py's _reset()/_rollback()
    documents) — browser-verified: this triggered _reset()'s "could not
    reset ... row" warning after a Run/document switch, and corrupted the
    ENTIRE prior session for the REST of the browser session (the
    bry17->bry18->bry19 chain rendered bry19 blank). A regular def function,
    however, has a REAL, isolated Python function scope (like
    cohort()/cumulative() above, never observed to leak) — the loop variable
    can never reach _shared_vars regardless of Brython version or dict
    implementation."""
    total = 0.0
    for row in rows:
        total += row["mill_nok"]
    return total

def column(rows, field):
    """Pick out a single column from the row list (same function-scope
    rationale as total_cost() above) — used by the table cell instead of a
    bare list comprehension directly at cell top level."""
    out = []
    for row in rows:
        out.append(row[field])
    return out

# Reference: total cost at 50% uptake (instead of the 10% default) — a
# natural "what if uptake turns out large" anchor. The delta arrow (down =
# good) shows whether the current setting is cheaper or more expensive than
# that.
REF_TOTAL = round(total_cost(cohort(5000, 200, 50)), 1)

#%% brython id=kontroller
import ui

price = ui.slider(5000, 50000, value=5000, step=5000, label="Price per patient (NOK)",
                   name="price", on_change=["kpi", "kurve", "tabell"], sync_to="price")
patients = ui.slider(200, 2000, value=200, step=200, label="Eligible patients",
                      name="patients", on_change=["kpi", "kurve", "tabell"], sync_to="patients")
uptake_pct = ui.slider(10, 90, value=10, step=10, label="Uptake (%)",
                        name="uptake_pct", on_change=["kpi", "kurve", "tabell"], sync_to="uptake_pct")

def _toggle_advanced(visible):
    w = ui.widget("uptake_pct")
    if w is None:
        return
    if visible:
        w.show()
    else:
        w.hide()

advanced = ui.checkbox(label="Advanced (show uptake rate)", value=True, name="advanced",
                        on_change=_toggle_advanced)

#%% brython id=kpi cols=2
import ui

rows = cohort(price, patients, uptake_pct)
show(ui.kpi(round(total_cost(rows), 1), unit="mill. NOK",
            label="Total cost, 5 years", fmt=",.1f", ref=REF_TOTAL, bra="ned"),
     ui.kpi(rows[-1]["patients"], unit="pcs", label="Patients in year 5"))

#%% brython id=kurve
import plotly_express_brython as pe

pe.line(x=YEARS, y=cumulative(cohort(price, patients, uptake_pct)),
        title="Cumulative cost (mill. NOK)")

#%% brython id=tabell
import pandas_brython as pd

rows = cohort(price, patients, uptake_pct)
pd.DataFrame({"Year": YEARS,
              "Patients": column(rows, "patients"),
              "Mill. NOK": column(rows, "mill_nok")})

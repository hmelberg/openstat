# Budsjettvirkning — nytt legemiddel

Tre glidebrytere styrer tre mål-celler samtidig: `on_change=["kpi", "kurve",
"tabell"]` (én liste, TRE mål) + `sync_to=` på hver av dem. Avkrysningsboksen
«Avansert» bruker i stedet `ui.widget("opptak_pst")` — HÅNDTAKET til en
allerede deklarert kontroll — og kaller `.hide()`/`.show()` på den. Det er en
push-callback (ingen rerun): den endrer bare synligheten til
opptak-glidebryteren, selve verdien lever videre uendret (`sync_to`) enten
den vises eller ei.

#%% brython
AAR = [1, 2, 3, 4, 5]

def kohort(pris, pasienter, opptak_pst):
    """Enkel opptakskurve: andelen naas gradvis over tre aar."""
    rader = []
    for t in AAR:
        n = round(pasienter * min(1.0, (opptak_pst / 100) * t / 3))
        rader.append({"aar": t, "pasienter": n, "mill_kr": round(n * pris / 1e6, 2)})
    return rader

def kumulativ(rader):
    """Lokal funksjonsscope med vilje (unngaar lekkasje av løkkevariabelen
    til sesjonsglobalene - se total_kostnad() under for detaljene rundt
    HVORFOR: exit gate-funn 2026-07-18, browser-verifisert)."""
    kum, s = [], 0.0
    for rad in rader:
        s += rad["mill_kr"]
        kum.append(round(s, 2))
    return kum

def total_kostnad(rader):
    """Sum av mill_kr over en radliste. EN EKTE FUNKSJON med vilje - IKKE
    en bar generator-/comprehension-uttrykk direkte på celletoppnivå.
    Exit gate-funn (2026-07-18): en bar generator-uttrykk på celletoppnivå
    (f.eks. `sum(rad["mill_kr"] for rad in rader)` rett i en celle) kan
    lekke løkkevariabelen `rad` inn i _shared_vars via en Brython 3.12
    dict-felle (samme felle som brython_runner.py sin _reset()/_rollback()
    dokumenterer) - browser-verifisert: dette utløste _reset()s
    "kunne ikke tilbakestille ... rad"-varsel etter en «Kjør»/dokumentbytte,
    og korrumperte tidligere HELE sesjonen for RESTEN av nettleserøkta
    (bry17->bry18->bry19-kjeden rendret bry19 blankt). En vanlig def-
    funksjon har derimot en EKTE, isolert Python-funksjonsscope (som
    kohort()/kumulativ() over, aldri observert å lekke) - løkkevariabelen
    kan aldri nå _shared_vars uansett Brython-versjon eller dict-implementasjon."""
    tot = 0.0
    for rad in rader:
        tot += rad["mill_kr"]
    return tot

def kolonne(rader, felt):
    """Plukk ut én kolonne fra radlisten (samme funksjonsscope-begrunnelse
    som total_kostnad() over) - brukt av tabell-cellen i stedet for en
    bar list comprehension direkte på celletoppnivå."""
    ut = []
    for rad in rader:
        ut.append(rad[felt])
    return ut

# Referanse: totalkostnad ved 50 % opptak (i stedet for default 10 %) - et
# naturlig "hva om opptaket blir stort"-anker. Delta-pilen (ned = bra) viser
# om dagens innstilling er billigere eller dyrere enn det.
REF_TOTAL = round(total_kostnad(kohort(5000, 200, 50)), 1)

#%% brython id=kontroller
import ui

pris = ui.slider(5000, 50000, value=5000, step=5000, label="Pris per pasient (kr)",
                  name="pris", on_change=["kpi", "kurve", "tabell"], sync_to="pris")
pasienter = ui.slider(200, 2000, value=200, step=200, label="Aktuelle pasienter",
                       name="pasienter", on_change=["kpi", "kurve", "tabell"], sync_to="pasienter")
opptak_pst = ui.slider(10, 90, value=10, step=10, label="Opptak (%)",
                        name="opptak_pst", on_change=["kpi", "kurve", "tabell"], sync_to="opptak_pst")

def _veksle_avansert(vis):
    w = ui.widget("opptak_pst")
    if w is None:
        return
    if vis:
        w.show()
    else:
        w.hide()

avansert = ui.checkbox(label="Avansert (vis opptaksrate)", value=True, name="avansert",
                        on_change=_veksle_avansert)

#%% brython id=kpi cols=2
import ui

rader = kohort(pris, pasienter, opptak_pst)
show(ui.kpi(round(total_kostnad(rader), 1), unit="mill. kr",
            label="Totalkostnad 5 aar", fmt=",.1f", ref=REF_TOTAL, bra="ned"),
     ui.kpi(rader[-1]["pasienter"], unit="stk", label="Pasienter i aar 5"))

#%% brython id=kurve
import plotly_express_brython as pe

pe.line(x=AAR, y=kumulativ(kohort(pris, pasienter, opptak_pst)),
        title="Kumulativ kostnad (mill. kr)")

#%% brython id=tabell
import pandas_brython as pd

rader = kohort(pris, pasienter, opptak_pst)
pd.DataFrame({"Aar": AAR,
              "Pasienter": kolonne(rader, "pasienter"),
              "Mill. kr": kolonne(rader, "mill_kr")})

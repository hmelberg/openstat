## Sales 2026

Same sales figures as before, now as `#%%` cells: the KPI tiles live in one
`cols=2` cell (`ui.kpi` + `show(...)` to mount both cards in the same cell),
and the two chart cards below each have THEIR OWN region/cumulative control
— `rerun="self"` (the default) means each card reruns itself when its own
control changes, just like a dash card with its own widget strip.

#%% brython
import plotly_express_brython as pe

months = ["Jan", "Feb", "Mar", "Apr", "May", "Jun"]
sales = {"North": [120, 135, 150, 160, 172, 190],
         "South": [200, 210, 190, 240, 260, 255]}

def curve(region, cumulative, title):
    values = sales[region]
    if cumulative:
        out = []
        s = 0
        for v in values:
            s += v
            out.append(s)
        values = out
    return pe.line(x=months, y=values, title=title)

#%% brython id=kpi cols=2
import ui

show(ui.kpi(sum(sales["North"]) + sum(sales["South"]), unit="pcs", label="Total sales"),
     ui.kpi(max(sales["South"]), unit="pcs", label="Best month (South)"))

#%% brython id=plot1
import ui

region = ui.dropdown(["North", "South"], label="Region", name="region1")
cumulative = ui.checkbox(label="Cumulative", value=False, name="cumulative1")
curve(region, cumulative, "Trend")

#%% brython id=plot2
import ui

region = ui.dropdown(["North", "South"], label="Region", name="region2")
cumulative = ui.checkbox(label="Cumulative", value=False, name="cumulative2")
curve(region, cumulative, "Trend (copy with its own controls)")

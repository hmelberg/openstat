## Salg 2026

Samme salgstall som før, nå som `#%%`-celler: KPI-flisene ligger i én
`cols=2`-celle (`ui.kpi` + `show(...)` for å montere begge kortene i samme
celle), og de to kurve-kortene under har HVER SIN region-/kumulativt-kontroll
— `rerun="self"` (standarden) betyr at hvert kort kjører seg selv på nytt når
egen kontroll endres, akkurat som et dash-kort med sin egen widget-stripe.

#%% brython
import plotly_express_brython as pe

mnd = ["jan", "feb", "mar", "apr", "mai", "jun"]
salg = {"Nord": [120, 135, 150, 160, 172, 190],
        "Soer": [200, 210, 190, 240, 260, 255]}

def kurve(region, kumulativ, tittel):
    verdier = salg[region]
    if kumulativ:
        ut = []
        s = 0
        for v in verdier:
            s += v
            ut.append(s)
        verdier = ut
    return pe.line(x=mnd, y=verdier, title=tittel)

#%% brython id=kpi cols=2
import ui

show(ui.kpi(sum(salg["Nord"]) + sum(salg["Soer"]), unit="stk", label="Totalt salg"),
     ui.kpi(max(salg["Soer"]), unit="stk", label="Beste maaned (Soer)"))

#%% brython id=plot1
import ui

region = ui.dropdown(["Nord", "Soer"], label="Region", name="region1")
kumulativ = ui.checkbox(label="Kumulativt", value=False, name="kumulativ1")
kurve(region, kumulativ, "Utvikling")

#%% brython id=plot2
import ui

region = ui.dropdown(["Nord", "Soer"], label="Region", name="region2")
kumulativ = ui.checkbox(label="Kumulativt", value=False, name="kumulativ2")
kurve(region, kumulativ, "Utvikling (kopi med egne kontroller)")

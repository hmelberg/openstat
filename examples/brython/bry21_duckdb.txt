# label: duckdb — SQL in Brython (.sql → .df)
# Example: SQL in Brython mode — duckdb.sql(...) and .df()
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris
import duckdb
import plotly_express_brython as pe

# A dataset loaded with `# load` can be queried directly with SQL:
show(duckdb.sql("""
    SELECT species,
           round(avg(sepal_length), 2) AS mean_length,
           count(*) AS n
    FROM iris
    GROUP BY species
    ORDER BY mean_length DESC
"""))

# .df() gives a pandas DataFrame you can keep working with:
df = duckdb.sql("SELECT species, sepal_length, petal_length FROM iris").df()
pe.scatter(df, x="sepal_length", y="petal_length", color="species",
           title="Iris via SQL → pandas → plotly")

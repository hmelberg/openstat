#%% md
# Widgets in a Brython notebook

Brython runs on the main thread, so the `ui` module uses the same pull
model as Python mode: every `ui.*` call reads the control's current value,
and changes rerun the cell (or a named cell via `rerun=`).

#%% brython
import ui

n = ui.slider(1, 20, value=5, label="Number of rows")
species = ui.dropdown(sorted(set(iris["species"])), label="Species")

iris[iris["species"] == species].head(n)

#%% brython id=plot
# This cell USES n/species from the cell above (shared session), but has no
# widgets of its own — the ui.run_button below reruns it (target="plot").
import pandas_brython as pd
import plotly_express_brython as pe

sub = iris[iris["species"] == species].head(n)
pe.scatter(sub, x="sepal_length", y="petal_length",
           title="Iris — " + species + ", first " + str(n) + " rows")

#%% brython
import ui

ui.run_button("Update", target="plot")

# Dashboard — fordelingsutforsker

Normalfordelingen N(mu, sigma): flytt kontrollene under, så oppdaterer
tetthetskurven, andelen over terskelen og 97,5 %-kvantilen seg sammen.
`terskel` har en innebygd avspillingsknapp (`ui.play`) som stepper verdien
automatisk. Kontrollcellen setter `on_change=["kpi", "plott"]` (en liste med
to celle-id-er) OG `sync_to=` på hver kontroll — samme oppskrift som
`py06_dashboard.txt`/`r09_dashboard.txt`, her i Brython-dialekten (kpi-cellen
bruker `show(...)` i stedet for et tuppel-uttrykk, se `03_dashboard.txt` for
hvorfor).

#%% brython id=kontroller
import ui

mu = ui.slider(-5, 5, value=0, step=1, label="Forventning (mu)", name="mu",
               on_change=["kpi", "plott"], sync_to="mu")
sigma = ui.slider(1, 4, value=1, step=1, label="Standardavvik (sigma)", name="sigma",
                   on_change=["kpi", "plott"], sync_to="sigma")
terskel = ui.play(-5, 10, value=-5, step=1, label="Terskel (spill av)", name="terskel",
                   on_change=["kpi", "plott"], sync_to="terskel")

#%% brython id=kpi cols=2
import ui
from scipy.stats import norm

show(ui.kpi(round(100 * float(norm.sf(terskel, loc=mu, scale=sigma)), 1),
            unit="%", label="P(X > terskel)"),
     ui.kpi(round(float(norm.ppf(0.975, loc=mu, scale=sigma)), 2),
            label="97,5 %-kvantilen"))

#%% brython id=plott
import plotly_express_brython as pe
from scipy.stats import norm

def xakse(mu, sigma):
    lo, hi = mu - 4 * sigma, mu + 4 * sigma
    n = 120
    return [lo + i * (hi - lo) / n for i in range(n + 1)]

def tetthet(xs, mu, sigma):
    ut = []
    for x in xs:
        ut.append(float(norm.pdf(x, loc=mu, scale=sigma)))
    return ut

xs = xakse(mu, sigma)
ys = tetthet(xs, mu, sigma)
fig = pe.line(x=xs, y=ys, title="Tetthet")
fig.add_vline(x=terskel)
fig

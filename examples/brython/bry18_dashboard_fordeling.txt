# Dashboard — distribution explorer

The normal distribution N(mu, sigma): move the controls below and the
density curve, the share above the threshold, and the 97.5% quantile all
update together. `threshold` has a built-in play button (`ui.play`) that
steps the value automatically. The control cell sets `on_change=["kpi",
"plott"]` (a list of two cell ids) AND `sync_to=` on each control — the
same recipe as `py06_dashboard.txt`/`r09_dashboard.txt`, here in the
Brython dialect (the kpi cell uses `show(...)` instead of a tuple
expression — see `03_dashboard.txt` for why).

#%% brython id=kontroller
import ui

mu = ui.slider(-5, 5, value=0, step=1, label="Mean (mu)", name="mu",
               on_change=["kpi", "plott"], sync_to="mu")
sigma = ui.slider(1, 4, value=1, step=1, label="Std dev (sigma)", name="sigma",
                   on_change=["kpi", "plott"], sync_to="sigma")
threshold = ui.play(-5, 10, value=-5, step=1, label="Threshold (play)", name="threshold",
                     on_change=["kpi", "plott"], sync_to="threshold")

#%% brython id=kpi cols=2
import ui
from scipy.stats import norm

show(ui.kpi(round(100 * float(norm.sf(threshold, loc=mu, scale=sigma)), 1),
            unit="%", label="P(X > threshold)"),
     ui.kpi(round(float(norm.ppf(0.975, loc=mu, scale=sigma)), 2),
            label="97.5% quantile"))

#%% brython id=plott
import plotly_express_brython as pe
from scipy.stats import norm

def x_axis(mu, sigma):
    lo, hi = mu - 4 * sigma, mu + 4 * sigma
    n = 120
    return [lo + i * (hi - lo) / n for i in range(n + 1)]

def density(xs, mu, sigma):
    out = []
    for x in xs:
        out.append(float(norm.pdf(x, loc=mu, scale=sigma)))
    return out

xs = x_axis(mu, sigma)
ys = density(xs, mu, sigma)
fig = pe.line(x=xs, y=ys, title="Density")
fig.add_vline(x=threshold);
fig

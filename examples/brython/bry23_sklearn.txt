# label: sklearn — KMeans, PCA, regression
# Example: machine learning in Brython mode — sklearn subset (pure Python)
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris
import plotly_express_brython as pe
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score, confusion_matrix

def to_str_column(values):
    out = []
    for k in values:
        out.append(str(k))
    return out

def column(matrix, idx):
    out = []
    for row in matrix:
        out.append(row[idx])
    return out

def flag_virginica(species_values):
    out = []
    for s in species_values:
        out.append(1 if s == "virginica" else 0)
    return out

X = iris[["sepal_length", "sepal_width", "petal_length", "petal_width"]]

# KMeans: find 3 clusters with no ground truth
km = KMeans(n_clusters=3, random_state=1).fit(X)
iris["cluster"] = to_str_column(km.labels_.tolist())
show(pe.scatter(iris, x="petal_length", y="petal_width", color="cluster",
                title="KMeans clusters (unsupervised)"))

# PCA: four measurements down to two principal components
pc = PCA(n_components=2).fit_transform(X).tolist()
iris["pc1"] = column(pc, 0)
iris["pc2"] = column(pc, 1)
show(pe.scatter(iris, x="pc1", y="pc2", color="species", title="PCA of iris"))

# Classification: is the flower virginica?
y = flag_virginica(iris["species"])
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=1)
model = LogisticRegression().fit(X_train, y_train)
pred = model.predict(X_test)
print("Accuracy:", round(accuracy_score(y_test, pred), 3))
print("Confusion matrix (rows = actual):")
print(confusion_matrix(y_test, pred).tolist())

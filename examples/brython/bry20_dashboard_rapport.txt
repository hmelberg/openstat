# load https://raw.githubusercontent.com/hmelberg/safestat/master/data/iris.csv as iris
import pandas_brython as pd
import plotly_express_brython as pe
import dash

d = dash.dashboard("Iris — datarapport")

d.add("## Iris-datasettet\nKlassikeren fra 1936: 150 blomster, tre arter, "
      "fire mal. Alt under er lagt til med en `add()`-linje per objekt — "
      "ingen layout angitt, auto-layouten ordner plasseringen.")

d.add(len(iris), title="Blomster", unit="stk")
d.add(len(iris["species"].unique()), title="Arter")
d.add(round(iris["sepal_length"].mean(), 2), title="Snitt sepal-lengde", unit="cm")

vc = iris["species"].value_counts()
d.add(pe.pie(values=list(vc.values), names=list(vc.index), title="Fordeling av arter"))

d.add(pe.histogram(iris, x="petal_length", title="Petal-lengde"))

d.add(iris.head(8), title="De forste radene")

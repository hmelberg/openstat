# label: Live data — World Bank API
# Example: Read from a LIVE public API — World Bank life expectancy (JSON).
# Brython runs in the page, so we use the browser's own HTTP (synchronous ajax).
import json
from browser import ajax
import pandas_brython as pd
import plotly_express_brython as pe

url = ("https://api.worldbank.org/v2/country/NOR;SWE;DNK;FIN/"
       "indicator/SP.DYN.LE00.IN?format=json&per_page=400")
req = ajax.Ajax()
req.open("GET", url, False)      # False = synchronous (ok for small requests)
req.send()
records = json.loads(req.text)[1]   # element 0 is pagination meta

# Build columns, keep the latest year per country:
latest = {}
for r in records:
    if r["value"] is not None and r["country"]["value"] not in latest:
        latest[r["country"]["value"]] = round(r["value"], 1)

life = pd.DataFrame({"land": list(latest.keys()),
                     "forventet_levealder": list(latest.values())})
show(life)

pe.bar(life, x="land", y="forventet_levealder",
       title="Forventet levealder — siste år (Verdensbanken)")

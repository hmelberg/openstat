# label: seaborn — statistical graphics
# Example: seaborn in Brython mode — statistical graphics in one line
# The sns functions draw on the same figure as plt, just like real seaborn.
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

np.random.seed(7)
data = {
    "income": np.random.normal(550.0, 80.0, 120).tolist(),
    "age":    np.random.uniform(25.0, 60.0, 120).tolist(),
    "region": (["North", "South", "East"] * 40),
}

# Histogram with color groups:
sns.histplot(data=data, x="income", hue="region", bins=15)
plt.title("Income distribution per region")
plt.show()

# Mean with uncertainty (as in seaborn: mean + CI):
sns.barplot(data=data, x="region", y="income")
plt.title("Mean income per region")
plt.show()

# Scatter with a regression line:
sns.regplot(data=data, x="age", y="income")
plt.title("Income vs age (OLS line)")
plt.show()

# Distribution per group:
sns.boxplot(data=data, x="region", y="income")
plt.show()

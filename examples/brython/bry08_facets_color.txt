# label: Facets — one panel per category
# Example: facets — one panel per category (facet_col)
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris
import pandas_brython as pd
import plotly_express_brython as pe

# Facets split the chart into one panel per value:
show(pe.scatter(iris, x="sepal_length", y="sepal_width",
                facet_col="species", color="species",
                title="Iris — one panel per species"))

# Color alone keeps everything in one panel:
pe.histogram(iris, x="petal_length", color="species",
             title="Petal length per species")

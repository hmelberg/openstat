# label: KPI cards + pie
# Example: KPI cards with pe.indicator — headline numbers for dashboards
# load https://raw.githubusercontent.com/hmelberg/openstat/main/data/iris.csv as iris
import pandas_brython as pd
import plotly_express_brython as pe

count = len(iris)
mean_val = round(iris["sepal_length"].mean(), 2)
max_val = iris["sepal_length"].max()

show(pe.indicator(value=count, title="Number of flowers"))
show(pe.indicator(value=mean_val, title="Mean sepal length"))
show(pe.indicator(value=max_val, title="Longest sepal"))

pe.pie(iris, names="species", title="Distribution of species")

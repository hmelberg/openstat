# Iris via DuckDB

`import duckdb` lever i «data»-cellen under — hver celle har sin egen
replay-trygge DuckDB-bro (samme mekanikk som `bry21_duckdb.txt`, bare
per-celle i stedet for per-skript): et cache-miss kjører sporringen
asynkront og cellen kjøres på nytt automatisk. Sporringen forhaandskjores
for HVER dropdown-verdi her, slik at selve kontrollendringen alltid treffer
cachen med det samme.

#%% brython id=data
import duckdb
import pandas_brython as pd

oversikt = duckdb.sql("""
    SELECT species,
           round(avg(sepal_length), 2) AS snitt_sepal,
           round(avg(petal_length), 2) AS snitt_petal,
           count(*) AS n
    FROM iris
    GROUP BY species
    ORDER BY snitt_sepal DESC
""").df()

ARTER = oversikt["species"].tolist()

def bygg_detalj(arter):
    d = {}
    for art in arter:
        d[art] = duckdb.sql(
            "SELECT sepal_length, petal_length FROM iris "
            "WHERE species = '" + art + "'").df()
    return d

detalj = bygg_detalj(ARTER)

totalt = int(duckdb.sql("SELECT count(*) AS n FROM iris").df()["n"][0])

#%% brython id=kontroller
import ui

art = ui.dropdown(ARTER, label="Art", name="art", on_change=["spred"], sync_to="art")

#%% brython id=kpi cols=2
import ui

show(ui.kpi(totalt, unit="stk", label="Blomster i basen"),
     ui.kpi(len(ARTER), label="Arter"))

#%% brython id=spred
import plotly_express_brython as pe

pe.scatter(detalj[art], x="sepal_length", y="petal_length",
           title="Sepal mot petal — " + art)

# load https://raw.githubusercontent.com/hmelberg/safestat/master/data/iris.csv as iris
import duckdb
import plotly_express_brython as pe
import dash

# 1) All SQL forst - for dashboard() lages. Forste kjoring replayes til
#    alle sporringer er i cache; da bygges dashboardet noyaktig en gang.
oversikt = duckdb.sql("""
    SELECT species,
           round(avg(sepal_length), 2) AS snitt_sepal,
           round(avg(petal_length), 2) AS snitt_petal,
           count(*) AS n
    FROM iris
    GROUP BY species
    ORDER BY snitt_sepal DESC
""").df()

ARTER = oversikt["species"].tolist()

# 2) Forhandskjor den parameteriserte sporringen for HVER dropdown-verdi.
#    Widget-endringer kan ikke vente pa DuckDB - men na er alt cachetreff.
detalj = {art: duckdb.sql(
              "SELECT sepal_length, petal_length FROM iris "
              "WHERE species = '" + art + "'").df()
          for art in ARTER}

totalt = int(duckdb.sql("SELECT count(*) AS n FROM iris").df()["n"][0])

# 3) Dashboardet - kortene leser bare fra de materialiserte resultatene.
d = dash.dashboard("Iris via DuckDB", layout="""
    antall antall arter  arter
    spred  spred  spred  tabell
    spred  spred  spred  tabell
""")

d.add(totalt, title="Blomster i basen", unit="stk", at="antall")
d.add(len(ARTER), title="Arter", at="arter")

d.controls(art=ARTER)

d.add(lambda art: pe.scatter(detalj[art], x="sepal_length", y="petal_length",
                             title="Sepal mot petal — " + art),
      at="spred")

d.add(oversikt, title="SQL-aggregering (GROUP BY)", at="tabell")

# Iris via DuckDB

`import duckdb` lives in the "data" cell below — each cell has its own
replay-safe DuckDB bridge (the same mechanism as `bry21_duckdb.txt`, just
per-cell instead of per-script): a cache miss runs the query asynchronously
and the cell reruns automatically. The query is pre-run for EVERY dropdown
value here, so the control change itself always hits the cache immediately.

#%% brython id=data
import duckdb
import pandas_brython as pd

overview = duckdb.sql("""
    SELECT species,
           round(avg(sepal_length), 2) AS mean_sepal,
           round(avg(petal_length), 2) AS mean_petal,
           count(*) AS n
    FROM iris
    GROUP BY species
    ORDER BY mean_sepal DESC
""").df()

SPECIES = overview["species"].tolist()

def build_detail(species_list):
    d = {}
    for species in species_list:
        d[species] = duckdb.sql(
            "SELECT sepal_length, petal_length FROM iris "
            "WHERE species = '" + species + "'").df()
    return d

detail = build_detail(SPECIES)

total = int(duckdb.sql("SELECT count(*) AS n FROM iris").df()["n"][0])

#%% brython id=kontroller
import ui

species = ui.dropdown(SPECIES, label="Species", name="species", on_change=["spred"], sync_to="species")

#%% brython id=kpi cols=2
import ui

show(ui.kpi(total, unit="pcs", label="Flowers in the dataset"),
     ui.kpi(len(SPECIES), label="Species"))

#%% brython id=spred
import plotly_express_brython as pe

pe.scatter(detail[species], x="sepal_length", y="petal_length",
           title="Sepal vs petal — " + species)

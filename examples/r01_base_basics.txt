# Eksempel: base R — frekvens og gjennomsnitt
# table() for frekvens, tapply() for gruppestatistikk.
df <- data.frame(
  kjonn   = c("Mann", "Kvinne", "Mann", "Kvinne", "Mann", "Kvinne"),
  inntekt = c(620000, 540000, 710000, 820000, 480000, 610000),
  alder   = c(34, 41, 29, 52, 38, 45)
)

table(df$kjonn)
tapply(df$inntekt, df$kjonn, mean)
summary(df$inntekt)
